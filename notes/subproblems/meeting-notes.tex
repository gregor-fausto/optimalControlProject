\begin{equation}
\begin{aligned}
\dot{P} & = -b(t)P(t) \\
\dot{V} & = (a(t) + b(t))P(t) \\
\dot{I} & = b(t)P(t) - c(t)I(t)\\
\dot{F} & = c(t)I(t) 
\end{aligned}
\end{equation}
subject to 

\begin{equation}
\begin{aligned}
& 0 \le a(t)P(t) + b(t)P(t) + c(t)I(t) \le V \\
& 0 \le a(t)+b(t),c(t) \le M
\end{aligned}
\end{equation}

\bigskip 

Let $a(t) = v(t)u(t), b(t) = v(t)(1-u(t))$ 

\begin{equation}
\begin{aligned}
\dot{P} & = -v(t)(1-u(t))P(t) \\
\dot{V} & = v(t)P(t) \\
\dot{I} & = v(t)(1-u(t))P(t) - c(t)I(t)\\
\dot{F} & = c(t)I(t) 
\end{aligned}
\end{equation}

subject to the constraints 
\begin{equation}
\begin{aligned}
& 0 \le v(t), c(t) \le M \\
& 0 \le v(t)P(t) + c(t)I(t) \le V \\
& 0 \le u(t) \le 1
\end{aligned}
\end{equation}
